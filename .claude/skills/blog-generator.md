# TITLE UNIQUENESS VERIFICATION

Before generating any post, verify uniqueness:

```javascript
// Pseudo-code for title verification
function isUnique(newTitle, existingTitles) {
  // Exact match check
  if (existingTitles.includes(newTitle)) return false;

  // Similarity check (no 80%+ similar titles)
  for (const existing of existingTitles) {
    if (similarity(newTitle, existing) > 0.8) return false;
  }

  return true;
}

// Location rotation to ensure variety
const usedLocations = new Map(); // topic -> [locations]
function getNextLocation(topic, preferredLocations) {
  const used = usedLocations.get(topic) || [];
  const available = preferredLocations.filter(l => !used.includes(l));
  if (available.length === 0) {
    // All locations used for this topic - vary the angle instead
    return { location: preferredLocations[0], varyAngle: true };
  }
  return { location: available[0], varyAngle: false };
```
